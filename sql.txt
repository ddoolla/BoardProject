* 회원 테이블 생성
create table member (
	uNum number(4) primary key,
	userId varchar2(20) not null,
	userPass varchar2(20) not null,
                  name varchar2(10) not null,
	phone varchar2(13) not null,
	email varchar2(30) not null
);
* 게시판 테이블 생성
create table board (
	cNum Number(4) primary key,
	uNum number(4) references member(uNum),
	title varchar2(50) not null,
	content varchar2(1000) not null,
	writeDate date default sysdate,
	visitNum number(4) default 0
);

* 회원수 시퀀스
create sequence seq_mem_num 
    increment by 1
    start with 1
    nocache;
 
* 콘텐츠수 시퀀스
create sequence seq_cont_num
    increment by 1
    start with 1
    nocache;
